ORG 1000h ;PARAMETROS, Memoria  de datos
  NUM1 DB 5h
  NUM2 DB 3h
  RES DW ?
ORG 3000h ;SUBRUTINAS
  RET  
ORG 2000h ;PROGRAMA PRINCIPAL
  MOV DX,0 ; USO DX PARA GUARDAR EL RES
  MOV AL,NUM1
  CMP AL,0 ;ME FIJO QUE EL NUMERO NO SEA 0, PARA NO MULTIPLICAR POR 0
  JZ FIN ; SI EL FLAG Z DA 1 ES PORQUE ES 0
  MOV AH,0 ;COMPLETO LA PARTE ALTA DE AX CON CERO PARA PODER OPERAR
  MOV CL, NUM2
  CMP CL,0 ;ME FIJO QUE EL NUMERO NO SEA 0, PARA NO MULTIPLICAR POR 0
  JZ FIN ; SI EL FLAG Z DA 1 ES PORQUE ES 0
  LOOP: ADD DX,AX ;SUMO EL NUMERO Y LO GUARDO EN DX
        DEC CL ;DECREMENTO CL, QUE ES LA CANTIDAD DE VECES QUE TENGO QUE MULTIPLICAR EL NUMERO
        JNZ LOOP ;SIGO HASTA LLEGAR A CERO
  FIN:MOV RES,DX ;GUARDO EL RESULTADO EN RES
  END
