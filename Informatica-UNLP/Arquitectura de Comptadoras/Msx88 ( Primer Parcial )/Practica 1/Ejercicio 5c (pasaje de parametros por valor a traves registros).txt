ORG 1000h ;PARAMETROS, Memoria  de datos
  A DW 5h
  B DW 3h
  C DW 4h
  D DW ?
ORG 3000h ;SUBRUTINAS
  SUBRUTINA: ADD AX,BX
             SUB AX,CX
             MOV DX,AX
  RET

  MOV AX,A
  MOV BX,B
  MOV CX,C
  CALL SUBRUTINA
  MOV D,DX
  HLT
  END
